# 楠 Kusunoki Mono ── コードと日本語がグリッドに揃う
from dataclasses import dataclass

@dataclass
class 商品:
    名前: str
    価格: int             # 税込、円
    在庫: bool = True     # ※ 取り寄せは False

def 合計(かご: list[商品]) -> int:
    """かごの合計金額を返す。空なら 0 円。"""
    return sum(x.価格 for x in かご if x.在庫)

かご = [商品("ほうじ茶", 700), 商品("最中", 380)]
assert 合計(かご) != 0 and 合計(かご) >= 500   # リガチャ: != >= ->
print(f"合計 {合計(かご):,} 円")               # => 1,080 円
